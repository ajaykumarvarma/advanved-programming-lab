//Find the repetative substrings in the given string(bag of words)
string=input("Enter a string:")
l=string.split(' ')
s=set(l)
dic={}
c=0
for i in s:
    c=0
    for j in l:
        if i==j:
            c+=1 
    dic[i]=c
for i in dic:
    if dic[i]>1:
        print(i,":",dic[i])
